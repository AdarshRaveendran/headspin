N=int(input("Enter a Number"))#take input from the user
for num in range(0,N):#lower limit 0 to upper limit N
  temp=num #num assign to temp
  sum=0 #initialize sum
  while temp>0: #find sum of the cube of  each digits
      rem=temp%10
      sum=sum+rem*rem*rem
      temp=temp//10
      if sum==num:
           print (num)
